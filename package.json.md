## Структура package.json

Файл package.json в проекте Vue (как и в любом Node.js-проекте) выполняет несколько ключевых функций:

- Содержит список всех библиотек и зависимостей, необходимых для проекта (например, Vue.js, Vue Router, Axios и т.д.).
- Указывает версии зависимостей (dependencies для продакшена, devDependencies для разработки), чтобы обеспечить совместимость и воспроизводимость.

```plaintext
{
  "name": "vue-documentation-playground", // Название проекта. Уникальный идентификатор, используется для идентификации проекта в npm (если публикуется) или просто для удобства. В данном случае — это просто имя для локального проекта.
  "version": "0.0.0", // Версия проекта. Следует семантическому версионированию (SemVer: major.minor.patch). "0.0.0" указывает на начальную версию, обычно для проектов в разработке.
  "private": true, // Указывает, что проект частный и не будет опубликован в npm. Предотвращает случайную публикацию и делает проект исключительно локальным/внутренним.
  "type": "module", // Указывает, что JavaScript-файлы в проекте используют ES Modules (ESM) с синтаксисом `import`/`export` вместо CommonJS (`require`/`module.exports`). Это важно для современных инструментов, таких как Vite.
  "scripts": { // Объект, содержащий пользовательские команды, которые можно запускать через `npm run <script>` или `yarn <script>`.
    "dev": "vite", // Запускает Vite в режиме разработки (development mode) с горячей перезагрузкой (hot module replacement).
    "build": "run-p type-check \"build-only {@}\" --", // Выполняет две команды параллельно (с помощью `npm-run-all2`): `type-check` (проверка типов с TypeScript) и `build-only` (сборка проекта). `{@}` передаёт аргументы.
    "preview": "vite preview", // Запускает локальный сервер для предварительного просмотра собранного приложения (обычно после `build`).
    "test:unit": "vitest", // Запускает юнит-тесты с помощью Vitest, современного тестового фреймворка, совместимого с Vite.
    "build-only": "vite build", // Выполняет сборку проекта для продакшена с помощью Vite (оптимизированный билд).
    "type-check": "vue-tsc --build", // Проверяет типы TypeScript в проекте с помощью `vue-tsc` (TypeScript-компилятор для Vue).
    "lint": "eslint . --fix", // Запускает ESLint для проверки и автоматического исправления ошибок кода во всех файлах проекта.
    "format": "prettier --write src/" // Форматирует код в папке `src/` с помощью Prettier, переписывая файлы в соответствии с правилами форматирования.
  },
  "dependencies": { // Зависимости, необходимые для работы приложения в продакшене.
    "vue": "^3.5.13" // Основной фреймворк Vue.js версии 3.5.13 или новее.
  },
  "devDependencies": { // Зависимости, необходимые только для разработки, тестирования и сборки (не включаются в продакшен-билд).
    "@tsconfig/node22": "^22.0.1", // Преднастроенный конфиг TypeScript для Node.js версии 22, упрощает настройку `tsconfig.json`.
    "@types/jsdom": "^21.1.7", // Типы TypeScript для библиотеки `jsdom`, используемой для симуляции DOM в тестах.
    "@types/node": "^22.14.0", // Типы TypeScript для Node.js, чтобы обеспечить автодополнение и проверку типов при использовании встроенных модулей Node.
    "@vitejs/plugin-vue": "^5.2.3", // Плагин Vite для поддержки Vue-файлов (`.vue`), необходим для компиляции однофайловых компонентов.
    "@vitest/eslint-plugin": "^1.1.39", // Плагин ESLint для интеграции с Vitest, помогает проверять тестовый код.
    "@vue/eslint-config-prettier": "^10.2.0", // Конфигурация ESLint для совместимости с Prettier, предотвращает конфликты между правилами линтинга и форматирования.
    "@vue/eslint-config-typescript": "^14.5.0", // Конфигурация ESLint для TypeScript в Vue-проектах, добавляет правила для TypeScript-кода.
    "@vue/test-utils": "^2.4.6", // Утилиты для тестирования Vue-компонентов, используются с Vitest для написания юнит-тестов.
    "@vue/tsconfig": "^0.7.0", // Базовая конфигурация `tsconfig.json` для Vue-проектов, упрощает настройку TypeScript.
    "eslint": "^9.22.0", // Линтер для проверки качества кода (JavaScript/TypeScript/Vue), помогает находить ошибки и поддерживать единый стиль.
    "eslint-plugin-vue": "~10.0.0", // Плагин ESLint для поддержки правил, специфичных для Vue (например, проверка синтаксиса в `.vue`-файлах).
    "jiti": "^2.4.2", // Библиотека для динамической загрузки ES Modules, используется для поддержки ESM в некоторых инструментах.
    "jsdom": "^26.0.0", // Эмулятор DOM для запуска тестов в среде Node.js, имитирует браузерную среду.
    "npm-run-all2": "^7.0.2", // Утилита для параллельного или последовательного выполнения нескольких npm-скриптов (используется в скрипте `build`).
    "prettier": "3.5.3", // Инструмент для автоматического форматирования кода, обеспечивает единый стиль.
    "typescript": "~5.8.0", // TypeScript для статической типизации и улучшения разработки, используется с `vue-tsc`.
    "vite": "^6.2.4", // Быстрый инструмент сборки и разработки, заменяет Webpack в современных Vue-проектах.
    "vite-plugin-vue-devtools": "^7.7.2", // Плагин Vite для интеграции с Vue Devtools, улучшает отладку компонентов Vue.
    "vitest": "^3.1.1", // Тестовый фреймворк, оптимизированный для Vite, используется для юнит-тестирования.
    "vue-tsc": "^2.2.8" // Компилятор TypeScript для Vue, проверяет типы и генерирует декларации для `.vue`-файлов.
  }
}
```
